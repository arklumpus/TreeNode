523  /// <param name="additionalDataToCopy">A stream containing additional data that will be copied into the binary file.</param>
524  public static void WriteAllTrees(IList<TreeNode> trees, Stream outputStream, bool keepOpen = false, Action<double> progressAction = null, Stream additionalDataToCopy = null)
525  {
526  Contract.Requires(trees != null);
527 
528  Dictionary<string, int> allNamesLookup = new Dictionary<string, int>();
529  List<string> allNamesLookupReverse = new List<string>();
530 
531  Dictionary<(string, bool), int> allAttributesLookup = new Dictionary<(string, bool), int>();
532  List<(string, bool)> allAttributesLookupReverse = new List<(string, bool)>();
533 
534  bool includeNamesPerTree = false;
535  bool includeAttributesPerTree = false;
536 
537  for (int i = 0; i < trees.Count; i++)
538  {
539  int prevNameCount = allNamesLookup.Count;
540  int prevAttributeCount = allAttributesLookup.Count;
541 
542  int count = 0;
543 
544  int maxAttributeCount = 0;
545 
546  foreach (TreeNode node in trees[i].GetChildrenRecursiveLazy())
547  {
548  if (!string.IsNullOrEmpty(node.Name))
549  {
550  count++;
551  if (allNamesLookup.TryAdd(node.Name, allNamesLookup.Count))
552  {
553  allNamesLookupReverse.Add(node.Name);
554  }
555  }
556 
557  maxAttributeCount = Math.Max(maxAttributeCount, node.Attributes.Count);
558 
559  foreach (KeyValuePair<string, object> kvp in node.Attributes)
560  {
561  bool isDouble = kvp.Value is double;
562  if (allAttributesLookup.TryAdd((kvp.Key, isDouble), allAttributesLookup.Count))
563  {
564  allAttributesLookupReverse.Add((kvp.Key, isDouble));
565  }
566  }
567  }
568 
569  if (prevNameCount != 0 && (allNamesLookup.Count - prevNameCount) * 2 > count)
570  {
571  includeNamesPerTree = true;
572  }
573 
574  if (prevAttributeCount != 0 && (allAttributesLookup.Count - prevAttributeCount) * 2 > maxAttributeCount)
575  {
576  includeAttributesPerTree = true;
577  }
578 
579  if (includeNamesPerTree && includeAttributesPerTree)
580  {
581  break;
582  }
583  }
584 
585  using BinaryWriter writer = new BinaryWriter(outputStream, Encoding.UTF8, keepOpen);
586 
587  writer.Write(new byte[] { (byte)'#', (byte)'T', (byte)'R', (byte)'E' });
588 
589  if (!includeNamesPerTree && !includeAttributesPerTree)
590  {
591  writer.Write((byte)0b00000011);
592  }
593  else if (!includeNamesPerTree && includeAttributesPerTree)
594  {
595  writer.Write((byte)0b00000001);
596  }
597  else if (includeNamesPerTree && !includeAttributesPerTree)
598  {
599  writer.Write((byte)0b00000010);
600  }
601  else
602  {
603  writer.Write((byte)0b00000000);
604  }
605 
606  if (!includeNamesPerTree)
607  {
608  writer.WriteInt(allNamesLookup.Count);
609 
610  for (int i = 0; i < allNamesLookup.Count; i++)
611  {
612  writer.WriteMyString(allNamesLookupReverse[i]);
613  }
614  }
615 
616  if (!includeAttributesPerTree)
617  {
618  writer.WriteInt(allAttributesLookup.Count);
619 
620  for (int i = 0; i < allAttributesLookup.Count; i++)
621  {
622  writer.WriteMyString(allAttributesLookupReverse[i].Item1);
623  writer.WriteInt(allAttributesLookupReverse[i].Item2 ? 2 : 1);
624  }
625  }
626 
627  long[] addresses = new long[trees.Count];
628 
629  for (int i = 0; i < trees.Count; i++)
630  {
631  writer.Flush();
632  addresses[i] = outputStream.Position;
633  writer.WriteTree(trees[i], !includeNamesPerTree, !includeAttributesPerTree, allNamesLookup, allAttributesLookup);
634 
635  double progress = Math.Max(0, Math.Min(1, (double)(i + 1) / trees.Count));
636 
637  progressAction?.Invoke(progress);
638  }
639 
640  if (additionalDataToCopy != null)
641  {
642  additionalDataToCopy.CopyTo(outputStream);
643  }
644 
645  writer.Flush();
646 
647  long labelAddress = outputStream.Position;
648 
649  writer.WriteInt(addresses.Length);
650 
651  for (int i = 0; i < addresses.Length; i++)
652  {
653  writer.Write(addresses[i]);
654  }
655
