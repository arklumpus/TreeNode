50 /// <summary>
51 /// Computes the Sackin index of the tree (sum of the leaf depths).
52 /// </summary>
53 /// <param name="model">If this is <see cref="NullHypothesis.None"/>, the raw Sackin index is returned. If this is <see cref="NullHypothesis.YHK"/> or <see cref="NullHypothesis.PDA"/>, the Sackin
54 /// index is normalised with respect to the corresponding null tree model (which makes scores comparable across trees of different sizes).</param>
55 /// <returns>The Sackin index of the tree, either as a raw value, or normalised according to the selected null tree model.</returns>
56 public double SackinIndex(NullHypothesis model = NullHypothesis.None)
57 {
58 List<double> leafDepths = new List<double>();
59
60 List<TreeNode> leaves = this.GetLeaves();
61
62 foreach (TreeNode leaf in leaves)
63 {
64 leafDepths.Add(leaf.GetDepth());
65 }
66
67 double averageLeafDepth = leafDepths.Average();
68
69 int sackinIndex = (int)leafDepths.Sum();
70
71 switch (model)
72 {
73 case NullHypothesis.None:
74 return sackinIndex;
75 case NullHypothesis.YHK:
76 return (sackinIndex - 2 * leaves.Count * (from el in Enumerable.Range(2, leaves.Count - 1) select 1.0 / el).Sum()) / leaves.Count;
77 case NullHypothesis.PDA:
78 return sackinIndex / Math.Pow(leaves.Count, 1.5);
79 }
80
81 return double.NaN;
82 }

100 /// <summary>
101 /// Computes the expected value of the Colless index under the YHK model.
102 /// </summary>
103 /// <param name="numberOfLeaves">The number of leaves in the tree.</param>
104 /// <returns>The expected value of the Colless index for a tree with the specified <paramref name="numberOfLeaves"/>.</returns>
105 /// <remarks>Proof in DOI: 10.1214/105051606000000547</remarks>
106 public static double GetCollessExpectationYHK(int numberOfLeaves)
107 {
108 static double tN(int n)
109 {
110 if (n % 2 == 0)
111 {
112 return (n - 2) / 4.0;
113 }
114 else
115 {
116 return (n - 1) * (n - 1) / (4.0 * n);
117 }
118 }
119
120 double sum = 0;
121
122 for (int k = 1; k < numberOfLeaves; k++)
123 {
124 sum += (k - 1 - 2 * tN(k)) / ((k + 1) * (k + 2));
125 }
126
127 return numberOfLeaves - 1 - 2 * tN(numberOfLeaves) + 2 * (numberOfLeaves + 1) * sum;
128 }

130 /// <summary>
131 /// Compute the Colless index of the tree.
132 /// </summary>
133 /// <param name="model">If this is <see cref="NullHypothesis.None"/>, the raw Colless index is returned. If this is <see cref="NullHypothesis.YHK"/> or <see cref="NullHypothesis.PDA"/>, the Colless
134 /// index is normalised with respect to the corresponding null tree model (which makes scores comparable across trees of different sizes).</param>
135 /// <param name="yhkExpectation">If <paramref name="model"/> is <see cref="NullHypothesis.YHK"/>, you can optionally use this parameter to provide a pre-computed value for the expected value of the
136 /// Colless index under the YHK model. This is useful to save time if you need to compute the Colless index of many trees with the same number of leaves. If this is <see cref="double.NaN"/>, the
137 /// expected value under the YHK model is computed by this method.</param>
138 /// <returns>The Colless index of the tree.</returns>
139 public double CollessIndex(NullHypothesis model = NullHypothesis.None, double yhkExpectation = double.NaN)
140 {
141 (int score, int leaves) = this.ComputeCollessInner();
142
143 switch (model)
144 {
145 case NullHypothesis.None:
146 return score;
147 case NullHypothesis.YHK:
148 if (double.IsNaN(yhkExpectation))
149 {
150 yhkExpectation = GetCollessExpectationYHK(leaves);
151 }
152 return (score - yhkExpectation) / leaves;
153 case NullHypothesis.PDA:
154 return score / Math.Pow(leaves, 1.5);
155 }
156
157 return double.NaN;
158 }

160 /// <summary>
161 /// Computes the number of cherries in the tree.
162 /// </summary>
163 /// <param name="model">If this is <see cref="NullHypothesis.None"/>, the raw number of cherries is returned. If this is <see cref="NullHypothesis.YHK"/> or <see cref="NullHypothesis.PDA"/>, the number
164 /// of cherries is normalised with respect to the corresponding null tree model (which makes scores comparable across trees of different sizes).</param>
165 /// <returns>The number of cherries in the tree.</returns>
166 /// <remarks>Proofs in DOI: 10.1016/S0025-5564(99)00060-7</remarks>
167 public double NumberOfCherries(NullHypothesis model = NullHypothesis.None)
168 {
169 List<TreeNode> leaves = this.GetLeaves();
170
171 int numberOfCherries = 0;
172
173 for (int i = 0; i < leaves.Count; i++)
174 {
175 if (leaves[i].Parent.Children.Count == 2 && leaves[i].Parent.Children[0].Children.Count == 0 && leaves[i].Parent.Children[1].Children.Count == 0)
176 {
177 numberOfCherries++;
178 }
179 }
180
181 numberOfCherries /= 2;
182
183 switch (model)
184 {
185 case NullHypothesis.None:
186 return numberOfCherries;
187
188 case NullHypothesis.YHK:
189 return (numberOfCherries - leaves.Count / 3.0) / Math.Sqrt(2.0 * leaves.Count / 45.0);
190
191 case NullHypothesis.PDA:
192 double mu = (double)leaves.Count * (leaves.Count - 1) / (2.0 * (2 * leaves.Count - 5));
193 double sigmaSq = (double)leaves.Count * (leaves.Count - 1) * (leaves.Count - 4) * (leaves.Count - 5) / (2.0 * (2 * leaves.Count - 5) * (2 * leaves.Count - 5) * (2 * leaves.Count - 7));
194 return (numberOfCherries - mu) / Math.Sqrt(sigmaSq);
195 }
196
197 return double.NaN;
198 }
