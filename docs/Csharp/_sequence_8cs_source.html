60 /// <summary>
61 /// Creates a new <see cref="Sequence"/>.
62 /// </summary>
63 /// <param name="sequence">The sequence.</param>
64 public Sequence(ReadOnlySpan<char> sequence)
65 {
66 List<char> states = new List<char>();
67 Dictionary<char, int> characterIndices = new Dictionary<char, int>();
68
69 intSequence = new int[sequence.Length];
70
71 for (int i = 0; i < sequence.Length; i++)
72 {
73 if (sequence[i] == '-')
74 {
75 intSequence[i] = -1;
76 }
77 else if (!characterIndices.TryGetValue(sequence[i], out intSequence[i]))
78 {
79 characterIndices[sequence[i]] = states.Count;
80 states.Add(sequence[i]);
81 }
82 }
83
84 this.States = ImmutableArray.Create<char>(states.ToArray());
85 this.Conservation = null;
86 this.IndelProfile = null;
87 this.stringSequence = new string(sequence);
88 }

310 /// <summary>
311 /// Simulates the evolution of the <see cref="Sequence"/> over the specified amount of <paramref name="time"/>, using the specified rate matrix.
312 /// Insertions and deletions happen according to the specified <paramref name="indelModel"/>.
313 /// </summary>
314 /// <param name="rateMatrix">The rate matrix to simulate sequence evolution.</param>
315 /// <param name="indelModel">The insertion/deletion model.</param>
316 /// <param name="time">The length of time over which the sequence evolves.</param>
317 /// <param name="insertions">An array containing all the insertion events that have occurred during the evolution of the sequence (useful to
318 /// map the evolved sequence back onto the ancestral sequence).</param>
319 /// <returns>The evolved sequence and an array containing all the insertion events that happened during its evolution.</returns>
320 public Sequence Evolve(RateMatrix rateMatrix, IndelModel indelModel, double time, out Insertion[] insertions)
321 {
322 (int[] newSequence, int[][] insertionEvents, double[] positionRates, double[] gapProfile) = SequenceSimulation.Evolve(time, this.intSequence, rateMatrix.GetMatrix(), rateMatrix.GetEquilibriumFrequencies(), rateMatrix.GetExponential(), new double[] { indelModel.InsertionRate, indelModel.DeletionRate }, indelModel.InsertionSizeDistribution, indelModel.DeletionSizeDistribution, this.Conservation?.ToArray(), this.IndelProfile?.ToArray());
323
324 if (insertionEvents != null)
325 {
326 insertions = new Insertion[insertionEvents.Length];
327 for (int i = 0; i < insertionEvents.Length; i++)
328 {
329 insertions[i] = new Insertion(insertionEvents[i][0], insertionEvents[i][1]);
330 }
331 }
332 else
333 {
334 insertions = null;
335 }
336
337 return new Sequence(newSequence, rateMatrix.States.ToArray(), positionRates, gapProfile);
338 }
