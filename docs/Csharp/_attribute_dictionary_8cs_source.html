71 /// <summary>
72 /// Gets or sets the value of the attribute with the specified <paramref name="name"/>. Getting the value of attributes <c>"Name"</c>, <c>"Length"</c> and <c>"Support"</c> does not require a dictionary lookup.
73 /// </summary>
74 /// <param name="name">The name of the attribute to get/set.</param>
75 /// <returns>The value of the attribute, boxed into an <c>object</c>.</returns>
76 public object this[string name]
77 {
78 get
79 {
80 Contract.Requires(name != null);
81 if (name.Equals("Name", StringComparison.OrdinalIgnoreCase))
82 {
83 return _name;
84 }
85 else if (name.Equals("Length", StringComparison.OrdinalIgnoreCase))
86 {
87 return _length;
88 }
89 else if (name.Equals("Support", StringComparison.OrdinalIgnoreCase))
90 {
91 return _support;
92 }
93 else
94 {
95 return InternalStorage[name];
96 }
97 }
98
99 set
100 {
101 Contract.Requires(name != null);
102 InternalStorage[name] = value;
103
104 if (name.Equals("Name", StringComparison.OrdinalIgnoreCase))
105 {
106 _name = Convert.ToString(value, System.Globalization.CultureInfo.InvariantCulture);
107 }
108 else if (name.Equals("Length", StringComparison.OrdinalIgnoreCase))
109 {
110 _length = Convert.ToDouble(value, System.Globalization.CultureInfo.InvariantCulture);
111 }
112 else if (name.Equals("Support", StringComparison.OrdinalIgnoreCase))
113 {
114 _support = Convert.ToDouble(value, System.Globalization.CultureInfo.InvariantCulture);
115 }
116 }
